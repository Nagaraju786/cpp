#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

bool palindromecheck(string str)
{
  transform(str.begin(), str.end(), str.begin(), ::toupper); 
  string orgstr = str;
  reverse(str.begin(), str.end());
  
  if (orgstr == str)
  {
       
       return true;
  }
  else
  {
      
      return false;
  }
}

int main()
{
    cout << "Enter word or pharse" << endl;
    string str;
    cin >> str;
    
    if( palindromecheck(str) )
    {
        cout << "Given string is Palindrome" << endl;
    }
    else
    {
        cout << "Given string is Not Palindrome" << endl;
    }

    return 0;
}
