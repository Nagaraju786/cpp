#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

bool palindromecheck(string str)
{
     
  string orgstr = str;
  reverse(str.begin(), str.end());
  
  if (orgstr == str)
  {
       
       return true;
  }
  else
  {
      
      return false;
  }
}

int main()
{
    string str = "Step on no pets";
    if( palindromecheck(str) )
    {
        cout << "Given string is Palindrome" << endl;
    }
    else
    {
        cout << "Given string is Not Palindrome" << endl;
    }

    return 0;
}
